  this.calculate = package => {
    // ups calculations
    return 1.56
  }

  this.calculate = package => {
    // usps calculations

  }

const fedex = new Fedex()
const ups = new UPS()
const usps = new USPS()

const package = {
  from : "Lee",
  to: "John",
  weight: 4.6
}

function Shipping() {
  this.company;
  this.setStrategy = (company) => {
    this.company = company
  }
  this.calculate = (package) => {
    return this.company.calculate(package)
  }
}
